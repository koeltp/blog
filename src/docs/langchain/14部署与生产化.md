## 二、LangServe 快速部署

```bash
pip install langserve
```

```python
# serve.py
from fastapi import FastAPI
from langserve import add_routes
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser

app = FastAPI(title="客服机器人")

# 定义链
prompt = ChatPromptTemplate.from_messages([
    ("system", "你是客服助手。"),
    ("human", "{question}"),
])
chain = prompt | ChatOpenAI(model="gpt-4o-mini") | StrOutputParser()

# 自动生成 REST API + Playground
add_routes(app, chain, path="/chat")

# 启动：uvicorn serve:app --reload --port 8000
```

LangServe 自动提供：
- `POST /chat/invoke` — 同步调用
- `POST /chat/stream` — 流式调用
- `POST /chat/batch` — 批量调用
- `GET /chat/playground` — 交互式测试页面

## 三、FastAPI 生产部署

### 3.1 完整服务

```python
# server.py
import asyncio
from contextlib import asynccontextmanager
from fastapi import FastAPI, HTTPException
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from app.main import chat, rag_chain
from app.knowledge.vectorstore import init_vectorstore

@asynccontextmanager
async def lifespan(app: FastAPI):
    """应用启动/关闭"""
    init_vectorstore()  # 初始化知识库
    yield

app = FastAPI(title="客服机器人 API", lifespan=lifespan)

class ChatRequest(BaseModel):
    message: str
    session_id: str = "default"

class ChatResponse(BaseModel):
    answer: str
    session_id: str

@app.post("/chat", response_model=ChatResponse)
async def chat_endpoint(req: ChatRequest):
    try:
        answer = await chat(req.message, req.session_id)
        return ChatResponse(answer=answer, session_id=req.session_id)
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.post("/chat/stream")
async def stream_endpoint(req: ChatRequest):
    """SSE 流式输出"""
    async def generate():
        async for chunk in rag_chain.astream(req.message):
            yield f"data: {chunk}\n\n"
        yield "data: [DONE]\n\n"

    return StreamingResponse(
        generate(),
        media_type="text/event-stream",
    )

@app.get("/health")
async def health():
    return {"status": "ok"}
```

### 3.2 前端对接流式输出

```javascript
// 前端调用流式接口
async function streamChat(message) {
    const resp = await fetch('/chat/stream', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ message, session_id: 'user-001' }),
    });

    const reader = resp.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        const text = decoder.decode(value);
        const lines = text.split('\n');
        for (const line of lines) {
            if (line.startsWith('data: ') && !line.includes('[DONE]')) {
                const chunk = line.slice(6);
                appendToChat(chunk);  // 追加到聊天界面
            }
        }
    }
}
```

## 四、Docker 容器化

```dockerfile
# Dockerfile
FROM python:3.12-slim

WORKDIR /app

# 安装依赖
COPY pyproject.toml .
RUN pip install --no-cache-dir .

# 复制代码
COPY app/ app/
COPY data/ data/
COPY server.py .

EXPOSE 8000

CMD ["uvicorn", "server:app", "--host", "0.0.0.0", "--port", "8000"]
```

```yaml
# docker-compose.yml
version: "3.8"
services:
  bot:
    build: .
    ports:
      - "8000:8000"
    env_file:
      - .env
    volumes:
      - ./data:/app/data
    restart: unless-stopped

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
```

```bash
# 构建并启动
docker-compose up -d

# 查看日志
docker-compose logs -f bot
```

### 5.2 超时与重试

```python
from langchain_openai import ChatOpenAI

llm = ChatOpenAI(
    model="gpt-4o-mini",
    timeout=30,           # 30秒超时
    max_retries=3,        # 重试3次
)

# 链级别重试
chain = (prompt | llm | parser).with_retry(
    stop_after_attempt=3,
    retry_if_exception_type=(TimeoutError, ValueError),
)
```

### 5.3 降级策略

```python
# 主模型失败时切换备用
primary = ChatOpenAI(model="gpt-4o")
fallback = ChatOpenAI(model="gpt-4o-mini")

chain = prompt | primary.with_fallbacks([fallback]) | parser

# 全部失败时返回兜底回复
FALLBACK_ANSWER = "抱歉，系统繁忙，请稍后再试或联系人工客服 400-xxx-xxxx。"
```

### 5.4 监控告警

```python
import logging
from datetime import datetime

logger = logging.getLogger("customer_service")

class MetricsMiddleware:
    """简单的指标收集中间件"""

    def __init__(self):
        self.request_count = 0
        self.error_count = 0
        self.total_latency = 0

    def record(self, latency: float, success: bool):
        self.request_count += 1
        self.total_latency += latency
        if not success:
            self.error_count += 1

        # 错误率超过 10% 告警
        error_rate = self.error_count / self.request_count
        if error_rate > 0.1 and self.request_count > 10:
            logger.warning(f"错误率过高：{error_rate:.1%}")
```
