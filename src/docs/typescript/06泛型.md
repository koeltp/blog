# 泛型

> 上一章我们用面向对象封装了 TaskManager。本章学习 TypeScript 泛型——让类型成为参数，写出可以复用的通用代码，然后用它构建通用 Repository。

泛型是 TypeScript 最强大的特性之一。它让你写出既类型安全又通用的代码，避免 `any` 和类型断言。

## 为什么需要泛型

没有泛型，你要么丢失类型信息，要么重复代码：

```typescript
// ❌ 方式一：丢失类型
function identity(value: any): any {
  return value
}
const result = identity('hello')  // 类型为 any，没有提示

// ❌ 方式二：为每种类型写一个函数
function identityString(value: string): string { return value }
function identityNumber(value: number): number { return value }
```

泛型让类型成为参数：

```typescript
function identity<T>(value: T): T {
  return value
}

const s = identity('hello')   // 类型为 string
const n = identity(42)        // 类型为 number

// 显式指定类型参数（通常不需要，TS 能推断）
const b = identity<boolean>(true)
```

## 泛型函数

### 多类型参数

```typescript
function map<T, U>(arr: T[], fn: (item: T) => U): U[] {
  return arr.map(fn)
}

const lengths = map(['hello', 'world'], s => s.length)
// lengths: number[]
```

### 常用泛型函数模式

```typescript
// 过滤
function filter<T>(arr: T[], predicate: (item: T) => boolean): T[] {
  return arr.filter(predicate)
}

// 分组
function groupBy<T, K extends string>(arr: T[], keyFn: (item: T) => K): Record<K, T[]> {
  return arr.reduce((acc, item) => {
    const key = keyFn(item)
    if (!acc[key]) acc[key] = []
    acc[key].push(item)
    return acc
  }, {} as Record<K, T[]>)
}

// 去重
function uniqueBy<T, K>(arr: T[], keyFn: (item: T) => K): T[] {
  const seen = new Set<K>()
  return arr.filter(item => {
    const key = keyFn(item)
    if (seen.has(key)) return false
    seen.add(key)
    return true
  })
}
```

## 泛型约束

### extends — 限制类型参数的范围

```typescript
// T 必须有 length 属性
function logLength<T extends { length: number }>(value: T): void {
  console.log(value.length)
}

logLength('hello')     // ✅ string 有 length
logLength([1, 2, 3])   // ✅ 数组有 length
// logLength(42)       // ❌ number 没有 length

// 约束为特定接口
interface HasId {
  id: number
}

function findById<T extends HasId>(items: T[], id: number): T | undefined {
  return items.find(item => item.id === id)
}
```

### keyof — 约束为对象的键

```typescript
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key]
}

const task = { id: 1, title: '学习', status: 'todo' }

const id = getProperty(task, 'id')      // number
const title = getProperty(task, 'title') // string
// getProperty(task, 'name')             // ❌ 'name' 不是 task 的键
```

### new — 约束为可实例化的类型

```typescript
function createInstance<T>(CTOR: new (...args: any[]) => T, ...args: any[]): T {
  return new CTOR(...args)
}

class Task {
  constructor(public title: string) {}
}

const task = createInstance(Task, '学习 TS')  // Task
```

## 泛型接口

```typescript
// 通用仓库接口
interface Repository<T extends HasId> {
  findById(id: number): T | undefined
  findAll(): T[]
  add(item: T): void
  remove(id: number): boolean
  update(id: number, changes: Partial<T>): T | undefined
}

// 分页结果
interface PageResult<T> {
  items: T[]
  total: number
  page: number
  pageSize: number
}

// 事件发射器
interface EventEmitter<Events extends Record<string, any>> {
  on<K extends keyof Events>(event: K, listener: (payload: Events[K]) => void): void
  emit<K extends keyof Events>(event: K, payload: Events[K]): void
  off<K extends keyof Events>(event: K, listener: (payload: Events[K]) => void): void
}
```

## 泛型类

```typescript
class DataStore<T> {
  private items: Map<number, T> = new Map()
  private nextId = 1

  add(item: Omit<T, 'id'> & { id?: number }): T & { id: number } {
    const id = item.id ?? this.nextId++
    const record = { ...item, id } as T & { id: number }
    this.items.set(id, record)
    return record
  }

  findById(id: number): T | undefined {
    return this.items.get(id)
  }

  findAll(): T[] {
    return [...this.items.values()]
  }

  remove(id: number): boolean {
    return this.items.delete(id)
  }
}

// 使用
interface User { id: number; name: string; email: string }
const userStore = new DataStore<User>()
userStore.add({ name: 'Alice', email: 'a@b.com' })
```

## 条件类型

条件类型让类型可以"做判断"：

```typescript
// 基本语法：T extends U ? X : Y
type IsString<T> = T extends string ? true : false

type A = IsString<string>  // true
type B = IsString<number>  // false
```

### infer — 在条件类型中提取类型

```typescript
// 提取函数返回值类型
type MyReturnType<T> = T extends (...args: any[]) => infer R ? R : never

type R1 = MyReturnType<() => string>        // string
type R2 = MyReturnType<(x: number) => boolean>  // boolean

// 提取 Promise 内部类型
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T

type P1 = UnwrapPromise<Promise<string>>  // string
type P2 = UnwrapPromise<number>            // number

// 提取数组元素类型
type ElementOf<T> = T extends (infer E)[] ? E : never

type E1 = ElementOf<string[]>  // string
type E2 = ElementOf<number[]>  // number

// 提取构造函数参数类型
type ConstructorParams<T> = T extends new (...args: infer P) => any ? P : never
```

### 分布式条件类型

当 `T` 是联合类型时，条件类型会分布式计算：

```typescript
type ToArray<T> = T extends any ? T[] : never

type Result = ToArray<string | number>
// 分布式计算：ToArray<string> | ToArray<number>
// 结果：string[] | number[]
```

阻止分布式计算，用 `[T]`：

```typescript
type ToArrayNonDist<T> = [T] extends [any] ? T[] : never

type Result2 = ToArrayNonDist<string | number>
// 结果：(string | number)[]
```

## 实战：通用 Repository

结合泛型、约束和条件类型，构建一个类型安全的通用 Repository：

```typescript
// ===== 基础类型 =====

interface Entity {
  id: number
}

type CreateDTO<T extends Entity> = Omit<T, 'id' | 'createdAt' | 'updatedAt'>
type UpdateDTO<T extends Entity> = Partial<Omit<T, 'id' | 'createdAt' | 'updatedAt'>>

// ===== 仓库接口 =====

interface IRepository<T extends Entity> {
  findById(id: number): T | undefined
  findAll(): T[]
  findWhere(predicate: (item: T) => boolean): T[]
  add(dto: CreateDTO<T>): T
  update(id: number, dto: UpdateDTO<T>): T | undefined
  remove(id: number): boolean
  count(): number
}

// ===== 通用实现 =====

class Repository<T extends Entity> implements IRepository<T> {
  protected items: Map<number, T> = new Map()
  protected nextId = 1

  findById(id: number): T | undefined {
    return this.items.get(id)
  }

  findAll(): T[] {
    return [...this.items.values()]
  }

  findWhere(predicate: (item: T) => boolean): T[] {
    return this.findAll().filter(predicate)
  }

  add(dto: CreateDTO<T>): T {
    const item = {
      ...dto,
      id: this.nextId++,
      createdAt: new Date(),
      updatedAt: new Date(),
    } as T
    this.items.set(item.id, item)
    return item
  }

  update(id: number, dto: UpdateDTO<T>): T | undefined {
    const existing = this.items.get(id)
    if (!existing) return undefined

    const updated = { ...existing, ...dto, updatedAt: new Date() } as T
    this.items.set(id, updated)
    return updated
  }

  remove(id: number): boolean {
    return this.items.delete(id)
  }

  count(): number {
    return this.items.size
  }
}

// ===== 业务实体 =====

interface Task extends Entity {
  title: string
  description: string
  status: 'todo' | 'in-progress' | 'done'
  priority: 'low' | 'medium' | 'high' | 'urgent'
  createdAt: Date
  updatedAt: Date
}

// ===== 使用 =====

const taskRepo = new Repository<Task>()

// add 的参数类型自动推导为 { title: string; description: string; ... }
const task = taskRepo.add({
  title: '学习泛型',
  description: '掌握 TypeScript 泛型',
  status: 'todo',
  priority: 'high',
})

// update 的参数是 Partial 的，可以只传部分字段
taskRepo.update(task.id, { status: 'in-progress' })

// findWhere 的 predicate 参数类型为 Task
const urgentTasks = taskRepo.findWhere(t => t.priority === 'urgent')
```
