# 异步编程

> 上一章我们学会了用模块组织多文件项目。本章深入 TypeScript 的异步编程——Promise、async/await、泛型 Promise、并发控制，实现 Task Manager 的异步文件持久化。

JavaScript 是单线程语言，异步是处理 I/O 操作的核心方式。TypeScript 在 Promise 基础上增加了完整的泛型类型支持，让异步代码既优雅又类型安全。

## Promise 基础

### 创建 Promise

```typescript
const promise = new Promise<string>((resolve, reject) => {
  setTimeout(() => {
    resolve('完成')
    // reject(new Error('失败'))
  }, 1000)
})
```

`Promise<string>` 表示这个 Promise resolve 后的值类型为 `string`。

### 链式调用

```typescript
function fetchUser(id: number): Promise<{ name: string }> {
  return Promise.resolve({ name: 'Alice' })
}

function fetchPosts(userName: string): Promise<string[]> {
  return Promise.resolve(['Post 1', 'Post 2'])
}

fetchUser(1)
  .then(user => fetchPosts(user.name))
  .then(posts => console.log(posts))
  .catch(error => console.error(error))
```

## async / await

async/await 是 Promise 的语法糖，让异步代码看起来像同步代码：

```typescript
async function getUserPosts(userId: number): Promise<string[]> {
  try {
    const user = await fetchUser(userId)       // user: { name: string }
    const posts = await fetchPosts(user.name)   // posts: string[]
    return posts
  } catch (error) {
    console.error('获取失败:', error)
    return []
  }
}
```

### 返回值类型

```typescript
// async 函数总是返回 Promise
async function greet(name: string): Promise<string> {
  return `Hello, ${name}!`
}

// 等价于
function greet(name: string): Promise<string> {
  return Promise.resolve(`Hello, ${name}!`)
}
```

### 顶层 await

在 ES Module 的顶层可以直接使用 `await`：

```typescript
// top-level await（仅在 ES Module 中可用）
const config = await loadConfig()
console.log(config)
```

## 泛型 Promise

TypeScript 的 Promise 是泛型的，`Promise<T>` 的 `T` 决定了 `await` 结果的类型：

```typescript
// 手动标注
async function loadData(): Promise<string[]> {
  return ['a', 'b', 'c']
}

const data = await loadData()  // data: string[]

// Promise.all — 保留每个 Promise 的类型
const [users, posts, comments] = await Promise.all([
  fetchUsers(),    // Promise<User[]>
  fetchPosts(),    // Promise<Post[]>
  fetchComments(), // Promise<Comment[]>
])
// users: User[], posts: Post[], comments: Comment[]
```

### Promise.allSettled

```typescript
const results = await Promise.allSettled([
  fetchUsers(),
  fetchPosts(),
])

results.forEach(result => {
  if (result.status === 'fulfilled') {
    console.log('成功:', result.value)
  } else {
    console.log('失败:', result.reason)
  }
})
```

## 并发控制

### 串行 vs 并行

```typescript
// ❌ 串行 — 总耗时 = 所有请求耗时之和
async function serial() {
  const a = await fetch('/api/a')  // 1s
  const b = await fetch('/api/b')  // 1s
  const c = await fetch('/api/c')  // 1s
  // 总耗时 3s
}

// ✅ 并行 — 总耗时 = 最慢的请求
async function parallel() {
  const [a, b, c] = await Promise.all([
    fetch('/api/a'),  // 1s
    fetch('/api/b'),  // 1s
    fetch('/api/c'),  // 1s
  ])
  // 总耗时 1s
}
```

### 限制并发数

```typescript
async function mapConcurrent<T, R>(
  items: T[],
  concurrency: number,
  fn: (item: T) => Promise<R>,
): Promise<R[]> {
  const results: R[] = []
  let index = 0

  async function worker(): Promise<void> {
    while (index < items.length) {
      const currentIndex = index++
      results[currentIndex] = await fn(items[currentIndex])
    }
  }

  const workers = Array.from({ length: Math.min(concurrency, items.length) }, () => worker())
  await Promise.all(workers)
  return results
}

// 最多同时 3 个请求
const data = await mapConcurrent(urls, 3, url => fetch(url).then(r => r.json()))
```

### 超时控制

```typescript
function withTimeout<T>(promise: Promise<T>, ms: number): Promise<T> {
  const timeout = new Promise<never>((_, reject) =>
    setTimeout(() => reject(new Error(`超时: ${ms}ms`)), ms)
  )
  return Promise.race([promise, timeout])
}

// 5秒超时
const data = await withTimeout(fetchData(), 5000)
```

### 重试

```typescript
async function retry<T>(
  fn: () => Promise<T>,
  options: { retries?: number; delay?: number } = {},
): Promise<T> {
  const { retries = 3, delay = 1000 } = options
  let lastError: unknown

  for (let i = 0; i <= retries; i++) {
    try {
      return await fn()
    } catch (error) {
      lastError = error
      if (i < retries) {
        await new Promise(resolve => setTimeout(resolve, delay * (i + 1)))
      }
    }
  }

  throw lastError
}

const data = await retry(() => fetchFromAPI(), { retries: 3, delay: 1000 })
```

## 错误处理

### try/catch

```typescript
async function safeFetch(url: string): Promise<string | null> {
  try {
    const response = await fetch(url)
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`)
    }
    return await response.text()
  } catch (error) {
    if (error instanceof Error) {
      console.error(error.message)
    }
    return null
  }
}
```

### Result 模式

比 try/catch 更函数式的错误处理：

```typescript
type Result<T, E = Error> =
  | { ok: true; value: T }
  | { ok: false; error: E }

async function tryCatch<T, E = Error>(promise: Promise<T>): Promise<Result<T, E>> {
  try {
    const value = await promise
    return { ok: true, value }
  } catch (error) {
    return { ok: false, error: error as E }
  }
}

// 使用
const result = await tryCatch(fetchData())
if (result.ok) {
  console.log(result.value)   // T
} else {
  console.error(result.error) // E
}
```

## 实战：异步文件持久化

为 Task Manager 添加异步文件存储：

```typescript
import { readFile, writeFile } from 'node:fs/promises'
import path from 'node:path'

interface Task {
  readonly id: number
  title: string
  status: 'todo' | 'in-progress' | 'done'
  priority: 'low' | 'medium' | 'high' | 'urgent'
  createdAt: string
  updatedAt: string
}

class TaskStorage {
  constructor(private filePath: string) {}

  async load(): Promise<Task[]> {
    try {
      const data = await readFile(this.filePath, 'utf-8')
      return JSON.parse(data) as Task[]
    } catch {
      return []
    }
  }

  async save(tasks: Task[]): Promise<void> {
    const dir = path.dirname(this.filePath)
    await writeFile(this.filePath, JSON.stringify(tasks, null, 2), 'utf-8')
  }
}

class AsyncTaskManager {
  #tasks: Task[] = []
  #nextId = 1
  #storage: TaskStorage

  constructor(storagePath: string) {
    this.#storage = new TaskStorage(storagePath)
  }

  async init(): Promise<void> {
    this.#tasks = await this.#storage.load()
    this.#nextId = this.#tasks.length > 0
      ? Math.max(...this.#tasks.map(t => t.id)) + 1
      : 1
  }

  async add(title: string, priority: Task['priority'] = 'medium'): Promise<Task> {
    const task: Task = {
      id: this.#nextId++,
      title,
      status: 'todo',
      priority,
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    }
    this.#tasks.push(task)
    await this.#persist()
    return task
  }

  async update(id: number, changes: Partial<Pick<Task, 'title' | 'status' | 'priority'>>): Promise<Task | null> {
    const index = this.#tasks.findIndex(t => t.id === id)
    if (index === -1) return null

    this.#tasks[index] = {
      ...this.#tasks[index],
      ...changes,
      updatedAt: new Date().toISOString(),
    }
    await this.#persist()
    return this.#tasks[index]
  }

  async remove(id: number): Promise<boolean> {
    const index = this.#tasks.findIndex(t => t.id === id)
    if (index === -1) return false

    this.#tasks.splice(index, 1)
    await this.#persist()
    return true
  }

  list(): readonly Task[] {
    return Object.freeze([...this.#tasks])
  }

  async #persist(): Promise<void> {
    await this.#storage.save(this.#tasks)
  }
}

// ===== 使用 =====

async function main(): Promise<void> {
  const manager = new AsyncTaskManager('./data/tasks.json')
  await manager.init()

  const task = await manager.add('学习异步编程', 'high')
  console.log('添加:', task)

  await manager.update(task.id, { status: 'in-progress' })
  console.log('列表:', manager.list())
}

main().catch(console.error)
```
