# 装饰器与元数据

> 上一章我们实现了异步文件持久化。本章学习 TypeScript 装饰器——一种声明式的元编程机制，配合 reflect-metadata 实现日志、权限控制等横切关注点。

装饰器是 TypeScript 对 TC39 装饰器提案的早期实现，广泛用于 NestJS、TypeORM、Angular 等框架。2023 年 TC39 已通过 Stage 3，TS 5.0+ 支持新标准装饰器。

## 启用装饰器

在 `tsconfig.json` 中启用：

```json
{
  "compilerOptions": {
    "experimentalDecorators": true,
    "emitDecoratorMetadata": true
  }
}
```

> `emitDecoratorMetadata` 需要安装 `reflect-metadata`：`npm install reflect-metadata`。

## 类装饰器

类装饰器在类声明时执行，可以修改或替换类定义：

```typescript
function Logged<T extends { new (...args: any[]): any }>(constructor: T) {
  return class extends constructor {
    constructor(...args: any[]) {
      console.log(`创建实例: ${constructor.name}`)
      super(...args)
    }
  }
}

@Logged
class Task {
  constructor(public title: string) {}
}

const task = new Task('学习装饰器')
// 输出: 创建实例: Task
```

## 方法装饰器

方法装饰器接收三个参数：目标对象、方法名、属性描述符：

```typescript
function Log(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
  const originalMethod = descriptor.value

  descriptor.value = function (...args: any[]) {
    console.log(`调用: ${propertyKey}(${args.join(', ')})`)
    const result = originalMethod.apply(this, args)
    console.log(`返回: ${JSON.stringify(result)}`)
    return result
  }
}

class Calculator {
  @Log
  add(a: number, b: number): number {
    return a + b
  }
}

const calc = new Calculator()
calc.add(1, 2)
// 调用: add(1, 2)
// 返回: 3
```

### 异步方法装饰器

```typescript
function AsyncLog(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
  const originalMethod = descriptor.value

  descriptor.value = async function (...args: any[]) {
    const start = Date.now()
    console.log(`→ ${propertyKey} 开始`)
    try {
      const result = await originalMethod.apply(this, args)
      const elapsed = Date.now() - start
      console.log(`← ${propertyKey} 完成 (${elapsed}ms)`)
      return result
    } catch (error) {
      const elapsed = Date.now() - start
      console.log(`✗ ${propertyKey} 失败 (${elapsed}ms):`, error)
      throw error
    }
  }
}
```

## 属性装饰器

```typescript
function Required(target: any, propertyKey: string) {
  // 在类的原型上存储元数据
  const requiredFields = Reflect.getMetadata('requiredFields', target.constructor) || []
  requiredFields.push(propertyKey)
  Reflect.defineMetadata('requiredFields', requiredFields, target.constructor)
}

class Task {
  @Required
  title!: string

  @Required
  priority!: string

  description?: string
}
```

## 装饰器工厂

用工厂函数接收参数，返回装饰器：

```typescript
function Validate(min: number, max: number) {
  return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value

    descriptor.value = function (...args: any[]) {
      // 验证第一个参数
      const value = args[0]
      if (typeof value !== 'number' || value < min || value > max) {
        throw new Error(`${propertyKey}: 参数必须在 ${min}~${max} 之间`)
      }
      return originalMethod.apply(this, args)
    }
  }
}

class ScoreManager {
  @Validate(0, 100)
  setScore(score: number): void {
    console.log(`分数: ${score}`)
  }
}

const mgr = new ScoreManager()
mgr.setScore(85)   // ✅
// mgr.setScore(150) // ❌ Error: setScore: 参数必须在 0~100 之间
```

## 装饰器组合

多个装饰器从下到上执行（类似函数调用栈）：

```typescript
function First() {
  console.log('First(): 工厂')
  return function () { console.log('First(): 装饰器') }
}

function Second() {
  console.log('Second(): 工厂')
  return function () { console.log('Second(): 装饰器') }
}

class Demo {
  @First()
  @Second()
  method() {}
}

// 输出:
// Second(): 工厂
// First(): 工厂
// Second(): 装饰器
// First(): 装饰器
```

执行顺序：工厂函数从下到上，装饰器从下到上。

## reflect-metadata

`reflect-metadata` 提供在运行时读写类型元数据的能力：

```typescript
import 'reflect-metadata'

const METADATA_KEY = {
  required: 'required',
  validate: 'validate',
  role: 'role',
}

// ===== 装饰器 =====

function Required(target: any, propertyKey: string) {
  const fields = Reflect.getMetadata(METADATA_KEY.required, target.constructor) || []
  fields.push(propertyKey)
  Reflect.defineMetadata(METADATA_KEY.required, fields, target.constructor)
}

function Role(role: string) {
  return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    Reflect.defineMetadata(METADATA_KEY.role, role, target, propertyKey)
  }
}

// ===== 运行时校验 =====

function validateRequired(instance: object): string[] {
  const ctor = instance.constructor
  const fields: string[] = Reflect.getMetadata(METADATA_KEY.required, ctor) || []
  const missing: string[] = []

  for (const field of fields) {
    if (instance[field as keyof typeof instance] === undefined) {
      missing.push(field)
    }
  }

  return missing
}

function checkRole(instance: object, method: string, userRole: string): boolean {
  const required = Reflect.getMetadata(METADATA_KEY.role, instance, method)
  if (!required) return true
  return userRole === required
}
```

## 实战：日志与权限装饰器

为 Task Manager 添加声明式日志和权限控制：

```typescript
import 'reflect-metadata'

// ===== 装饰器定义 =====

type AsyncMethod = (...args: any[]) => Promise<any>

function Logged(target: any, propertyKey: string, descriptor: PropertyDescriptor) {
  const original = descriptor.value as AsyncMethod

  descriptor.value = async function (...args: any[]) {
    const start = Date.now()
    console.log(`[LOG] → ${propertyKey}(${JSON.stringify(args)})`)
    try {
      const result = await original.apply(this, args)
      const ms = Date.now() - start
      console.log(`[LOG] ← ${propertyKey} 完成 (${ms}ms)`)
      return result
    } catch (error) {
      const ms = Date.now() - start
      console.error(`[LOG] ✗ ${propertyKey} 失败 (${ms}ms)`, error)
      throw error
    }
  }
}

function RequireRole(role: string) {
  return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value as AsyncMethod

    descriptor.value = async function (this: any, ...args: any[]) {
      const currentUser = this.currentUser
      if (currentUser?.role !== role) {
        throw new Error(`权限不足: 需要 ${role} 角色`)
      }
      return original.apply(this, args)
    }
  }
}

function Deprecated(message?: string) {
  return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const original = descriptor.value

    descriptor.value = function (...args: any[]) {
      console.warn(`[DEPRECATED] ${propertyKey} 已弃用${message ? `: ${message}` : ''}`)
      return original.apply(this, args)
    }
  }
}

// ===== 业务类 =====

interface User {
  name: string
  role: 'admin' | 'user'
}

class SecuredTaskManager {
  currentUser: User

  constructor(user: User) {
    this.currentUser = user
  }

  @Logged
  async listTasks(): Promise<string[]> {
    return ['任务1', '任务2', '任务3']
  }

  @RequireRole('admin')
  @Logged
  async deleteTask(id: number): Promise<boolean> {
    console.log(`删除任务 #${id}`)
    return true
  }

  @Deprecated('请使用 listTasks()')
  @Logged
  async getTasks(): Promise<string[]> {
    return this.listTasks()
  }
}

// ===== 测试 =====

async function demo(): Promise<void> {
  // 管理员
  const admin = new SecuredTaskManager({ name: 'Alice', role: 'admin' })
  await admin.listTasks()
  await admin.deleteTask(1)

  // 普通用户
  const user = new SecuredTaskManager({ name: 'Bob', role: 'user' })
  await admin.listTasks()
  try {
    await user.deleteTask(2)  // 抛出权限错误
  } catch (error) {
    console.error((error as Error).message)
  }

  // 弃用方法
  await admin.getTasks()
}

demo()
```

## TC39 新标准装饰器

TypeScript 5.0+ 支持新的标准装饰器语法（不需要 `experimentalDecorators`）：

```typescript
// 新标准：更简洁，但功能更受限
function logged<T extends (...args: any[]) => any>(
  target: T,
  context: ClassMethodDecoratorContext,
) {
  function replacement(this: any, ...args: any[]) {
    console.log(`调用: ${String(context.name)}`)
    return target.apply(this, args)
  }
  return replacement
}

class Example {
  @logged
  greet(name: string) {
    return `Hello, ${name}!`
  }
}
```

> 目前主流框架（NestJS、TypeORM）仍使用旧版装饰器。新标准装饰器生态还在建设中。
