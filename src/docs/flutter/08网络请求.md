## 一、Flutter 网络请求概览

Flutter 中网络请求的本质是 HTTP 客户端调用，核心选择有两个：

| | http | dio |
|---|------|-----|
| 维护方 | Dart 官方 | 社区（Flutter Chinese 团队） |
| 功能 | 基础 HTTP 请求 | 拦截器、取消、超时、FormData、重试 |
| 体积 | 小 | 较大 |
| 学习成本 | 低 | 中 |
| 适用场景 | 简单请求 | 生产级应用 |

> **建议**：学习阶段用 http 理解原理，生产项目用 dio 提高效率。

## 二、http 包：基础请求

### 2.1 安装与基本用法

```yaml
dependencies:
  http: ^1.2.0
```

```dart
import 'package:http/http.dart' as http;

// GET 请求
Future<List<Journal>> fetchJournals() async {
  final response = await http.get(
    Uri.parse('https://api.example.com/journals'),
    headers: {'Authorization': 'Bearer $token'},
  );

  if (response.statusCode == 200) {
    final List data = jsonDecode(response.body);
    return data.map((json) => Journal.fromJson(json)).toList();
  } else {
    throw Exception('加载失败: ${response.statusCode}');
  }
}

// POST 请求
Future<Journal> createJournal(Journal journal) async {
  final response = await http.post(
    Uri.parse('https://api.example.com/journals'),
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer $token',
    },
    body: jsonEncode(journal.toJson()),
  );

  if (response.statusCode == 201) {
    return Journal.fromJson(jsonDecode(response.body));
  } else {
    throw Exception('创建失败: ${response.statusCode}');
  }
}

// PUT / DELETE
Future<void> updateJournal(Journal journal) async {
  final response = await http.put(
    Uri.parse('https://api.example.com/journals/${journal.id}'),
    headers: {'Content-Type': 'application/json'},
    body: jsonEncode(journal.toJson()),
  );
  if (response.statusCode != 200) throw Exception('更新失败');
}

Future<void> deleteJournal(String id) async {
  final response = await http.delete(
    Uri.parse('https://api.example.com/journals/$id'),
  );
  if (response.statusCode != 204) throw Exception('删除失败');
}
```

## 三、dio：生产级 HTTP 客户端

### 3.1 安装与基本配置

```yaml
dependencies:
  dio: ^5.4.0
```

```dart
import 'package:dio/dio.dart';

// 全局 Dio 实例
final dio = Dio(BaseOptions(
  baseUrl: 'https://api.example.com',
  connectTimeout: const Duration(seconds: 10),
  receiveTimeout: const Duration(seconds: 30),
  headers: {
    'Content-Type': 'application/json',
  },
));
```

### 3.2 拦截器

拦截器是 dio 最强大的功能，可以在请求发出前和响应返回后统一处理：

```dart
// 认证拦截器 — 自动添加 Token
dio.interceptors.add(InterceptorsWrapper(
  onRequest: (options, handler) {
    final token = AuthService.instance.token;
    if (token != null) {
      options.headers['Authorization'] = 'Bearer $token';
    }
    handler.next(options);
  },
  onError: (error, handler) async {
    if (error.response?.statusCode == 401) {
      // Token 过期，尝试刷新
      try {
        final newToken = await AuthService.instance.refreshToken();
        error.requestOptions.headers['Authorization'] = 'Bearer $newToken';
        // 用新 Token 重试
        final response = await dio.fetch(error.requestOptions);
        handler.resolve(response);
      } catch (_) {
        handler.reject(error);
      }
    } else {
      handler.next(error);
    }
  },
));

// 日志拦截器（开发环境）
dio.interceptors.add(LogInterceptor(
  requestBody: true,
  responseBody: true,
));

// 缓存拦截器
dio.interceptors.add(InterceptorsWrapper(
  onRequest: (options, handler) {
    // 对 GET 请求添加缓存头
    if (options.method == 'GET') {
      options.extra['cacheKey'] = options.uri.toString();
    }
    handler.next(options);
  },
));
```

### 3.3 请求与响应

```dart
// GET
final response = await dio.get('/journals', queryParameters: {
  'category': 'tech',
  'page': 1,
  'limit': 20,
});

// POST
final response = await dio.post('/journals', data: {
  'title': '新日记',
  'content': '今天天气不错',
});

// 上传文件
final formData = FormData.fromMap({
  'title': '带图片的日记',
  'content': '看看这张照片',
  'image': await MultipartFile.fromFile(
    '/path/to/image.jpg',
    filename: 'photo.jpg',
  ),
});
final response = await dio.post('/journals', data: formData);

// 下载文件
await dio.download(
  'https://example.com/file.pdf',
  '/save/path/file.pdf',
  onReceiveProgress: (received, total) {
    print('下载进度: ${(received / total * 100).toStringAsFixed(1)}%');
  },
);

// 取消请求
final cancelToken = CancelToken();
dio.get('/journals', cancelToken: cancelToken);
// 取消
cancelToken.cancel('用户取消了请求');
```

### 3.4 错误处理

```dart
try {
  final response = await dio.get('/journals');
} on DioException catch (e) {
  switch (e.type) {
    case DioExceptionType.connectionTimeout:
      print('连接超时');
    case DioExceptionType.sendTimeout:
      print('发送超时');
    case DioExceptionType.receiveTimeout:
      print('接收超时');
    case DioExceptionType.badResponse:
      // 服务器返回了错误状态码
      final statusCode = e.response?.statusCode;
      final data = e.response?.data;
      print('服务器错误: $statusCode, $data');
    case DioExceptionType.cancel:
      print('请求被取消');
    case DioExceptionType.connectionError:
      print('网络连接失败');
    default:
      print('未知错误: ${e.message}');
  }
}
```

## 四、JSON 序列化

Flutter 没有 Java 的反射机制（为了性能），JSON 序列化需要手动或用代码生成。

### 4.1 手动序列化（小型项目）

```dart
class Journal {
  final String id;
  final String title;
  final String content;
  final String category;
  final int likes;
  final DateTime createdAt;

  const Journal({
    required this.id,
    required this.title,
    required this.content,
    this.category = '生活',
    this.likes = 0,
    DateTime? createdAt,
  }) : createdAt = createdAt ?? DateTime.now();

  // 从 JSON 创建
  factory Journal.fromJson(Map<String, dynamic> json) {
    return Journal(
      id: json['id'] as String,
      title: json['title'] as String,
      content: json['content'] as String,
      category: json['category'] as String? ?? '生活',
      likes: json['likes'] as int? ?? 0,
      createdAt: json['created_at'] != null
          ? DateTime.parse(json['created_at'] as String)
          : null,
    );
  }

  // 转为 JSON
  Map<String, dynamic> toJson() {
    return {
      'id': id,
      'title': title,
      'content': content,
      'category': category,
      'likes': likes,
      'created_at': createdAt.toIso8601String(),
    };
  }
}
```

### 4.2 json_serializable（推荐）

手动序列化在字段多的时候很痛苦。`json_serializable` 自动生成 fromJson/toJson：

```yaml
dependencies:
  json_annotation: ^4.9.0

dev_dependencies:
  build_runner: ^2.4.0
  json_serializable: ^6.8.0
```

```dart
import 'package:json_annotation/json_annotation.dart';

part 'journal.g.dart';  // 生成代码的文件

@JsonSerializable()
class Journal {
  final String id;
  final String title;
  final String content;

  @JsonKey(name: 'category_name', defaultValue: '生活')
  final String category;

  @JsonKey(defaultValue: 0)
  final int likes;

  @JsonKey(name: 'created_at')
  final DateTime createdAt;

  const Journal({
    required this.id,
    required this.title,
    required this.content,
    this.category = '生活',
    this.likes = 0,
    required this.createdAt,
  });

  factory Journal.fromJson(Map<String, dynamic> json) => _$JournalFromJson(json);
  Map<String, dynamic> toJson() => _$JournalToJson(this);
}
```

**生成代码：**

```bash
# 一次性生成
dart run build_runner build

# 监听文件变化自动生成
dart run build_runner watch
```

**@JsonKey 常用参数：**

| 参数 | 作用 |
|------|------|
| `name` | JSON 字段名与 Dart 属性名不同时映射 |
| `defaultValue` | JSON 中缺失该字段时的默认值 |
| `ignore` | 忽略该字段，不参与序列化 |
| `fromJson` | 自定义反序列化函数 |
| `toJson` | 自定义序列化函数 |

## 五、API 服务封装

把网络请求封装为独立的 Service 层，和 UI 解耦：

```dart
// services/api_client.dart
class ApiClient {
  static ApiClient? _instance;
  late final Dio _dio;

  ApiClient._() {
    _dio = Dio(BaseOptions(
      baseUrl: const String.fromEnvironment('API_BASE_URL',
          defaultValue: 'https://api.example.com'),
      connectTimeout: const Duration(seconds: 10),
      receiveTimeout: const Duration(seconds: 30),
    ));

    _dio.interceptors.addAll([
      _AuthInterceptor(),
      _ErrorInterceptor(),
      if (kDebugMode) LogInterceptor(requestBody: true, responseBody: true),
    ]);
  }

  static ApiClient get instance => _instance ??= ApiClient._();

  Future<List<Journal>> getJournals({String? category, int page = 1}) async {
    final response = await _dio.get('/journals', queryParameters: {
      if (category != null) 'category': category,
      'page': page,
      'limit': 20,
    });
    return (response.data as List).map((j) => Journal.fromJson(j)).toList();
  }

  Future<Journal> getJournal(String id) async {
    final response = await _dio.get('/journals/$id');
    return Journal.fromJson(response.data);
  }

  Future<Journal> createJournal(Journal journal) async {
    final response = await _dio.post('/journals', data: journal.toJson());
    return Journal.fromJson(response.data);
  }

  Future<Journal> updateJournal(Journal journal) async {
    final response = await _dio.put('/journals/${journal.id}', data: journal.toJson());
    return Journal.fromJson(response.data);
  }

  Future<void> deleteJournal(String id) async {
    await _dio.delete('/journals/$id');
  }
}

// 认证拦截器
class _AuthInterceptor extends Interceptor {
  @override
  void onRequest(RequestOptions options, RequestInterceptorHandler handler) {
    final token = AuthService.instance.token;
    if (token != null) {
      options.headers['Authorization'] = 'Bearer $token';
    }
    handler.next(options);
  }
}

// 错误拦截器
class _ErrorInterceptor extends Interceptor {
  @override
  void onError(DioException err, ErrorInterceptorHandler handler) {
    // 统一错误处理
    final apiError = ApiError.fromDioException(err);
    handler.reject(DioException(
      requestOptions: err.requestOptions,
      response: err.response,
      type: err.type,
      error: apiError,
    ));
  }
}

// 统一错误模型
class ApiError {
  final String message;
  final int? statusCode;
  final bool isNetworkError;

  const ApiError({
    required this.message,
    this.statusCode,
    this.isNetworkError = false,
  });

  factory ApiError.fromDioException(DioException e) {
    switch (e.type) {
      case DioExceptionType.connectionError:
      case DioExceptionType.connectionTimeout:
        return const ApiError(message: '网络连接失败', isNetworkError: true);
      case DioExceptionType.badResponse:
        final status = e.response?.statusCode;
        return ApiError(
          message: _messageForStatus(status),
          statusCode: status,
        );
      default:
        return ApiError(message: e.message ?? '未知错误');
    }
  }

  static String _messageForStatus(int? status) {
    return switch (status) {
      400 => '请求参数错误',
      401 => '未登录或登录已过期',
      403 => '没有权限',
      404 => '资源不存在',
      500 => '服务器内部错误',
      _ => '请求失败 ($status)',
    };
  }
}
```

## 六、小结

| 主题 | 核心要点 |
|------|---------|
| http | 官方包，简单够用，适合学习 |
| dio | 拦截器、取消、超时、FormData，生产级首选 |
| JSON 序列化 | 小项目手动，大项目用 json_serializable |
