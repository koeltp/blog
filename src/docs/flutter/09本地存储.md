## 一、本地存储方案对比

| 方案 | 数据类型 | 容量 | 适用场景 |
|------|---------|------|---------|
| SharedPreferences | 键值对 | KB 级 | 配置、标记、小数据 |
| 文件 | 任意 | 不限 | 图片、日志、导出数据 |
| SQLite | 结构化 | GB 级 | 业务数据、离线缓存 |
| Isar/Hive | 对象 | GB 级 | NoSQL 需求、高性能 |

## 二、SharedPreferences

### 2.1 基本用法

```yaml
dependencies:
  shared_preferences: ^2.2.0
```

```dart
import 'package:shared_preferences/shared_preferences.dart';

// 写入
Future<void> saveSettings() async {
  final prefs = await SharedPreferences.getInstance();
  await prefs.setString('username', 'Koeltp');
  await prefs.setInt('launch_count', 10);
  await prefs.setBool('dark_mode', true);
  await prefs.setStringList('tags', ['flutter', 'dart']);
}

// 读取
Future<void> loadSettings() async {
  final prefs = await SharedPreferences.getInstance();
  final username = prefs.getString('username') ?? '匿名';
  final launchCount = prefs.getInt('launch_count') ?? 0;
  final darkMode = prefs.getBool('dark_mode') ?? false;
  final tags = prefs.getStringList('tags') ?? [];
}

// 删除
await prefs.remove('username');

// 清空
await prefs.clear();
```

### 2.2 封装为 Service

```dart
class PreferencesService {
  static PreferencesService? _instance;
  late final SharedPreferences _prefs;

  PreferencesService._(this._prefs);

  static Future<PreferencesService> get instance async {
    _instance ??= PreferencesService._(await SharedPreferences.getInstance());
    return _instance!;
  }

  // 类型安全的访问器
  String get username => _prefs.getString('username') ?? '';
  set username(String v) => _prefs.setString('username', v);

  bool get darkMode => _prefs.getBool('dark_mode') ?? false;
  set darkMode(bool v) => _prefs.setBool('dark_mode', v);

  int get launchCount => _prefs.getInt('launch_count') ?? 0;
  Future<void> incrementLaunchCount() async {
    await _prefs.setInt('launch_count', launchCount + 1);
  }
}
```

## 三、文件读写

### 3.1 应用目录

```dart
import 'package:path_provider/path_provider.dart';

// 应用文档目录（可被 iCloud/备份同步）
final docDir = await getApplicationDocumentsDirectory();

// 临时目录（系统可能随时清理）
final tempDir = await getTemporaryDirectory();

// 应用支持目录（不适合用户可见的文件）
final supportDir = await getApplicationSupportDirectory();
```

### 3.2 读写文件

```dart
import 'dart:io';

// 写入文本文件
Future<void> writeJournalFile(String content) async {
  final dir = await getApplicationDocumentsDirectory();
  final file = File('${dir.path}/journals/latest.txt');
  await file.writeAsString(content);
}

// 读取文本文件
Future<String> readJournalFile() async {
  final dir = await getApplicationDocumentsDirectory();
  final file = File('${dir.path}/journals/latest.txt');
  if (await file.exists()) {
    return await file.readAsString();
  }
  return '';
}

// 写入 JSON 文件
Future<void> saveJournalsToFile(List<Journal> journals) async {
  final dir = await getApplicationDocumentsDirectory();
  final file = File('${dir.path}/journals/backup.json');
  final json = jsonEncode(journals.map((j) => j.toJson()).toList());
  await file.writeAsString(json);
}
```

## 四、SQLite + Drift

### 4.1 为什么用 Drift

直接用 SQLite 需要写大量 SQL 和手动映射，Drift 是 Flutter 最成熟的 SQLite ORM，提供类型安全的数据库操作。

```yaml
dependencies:
  drift: ^2.16.0
  sqlite3_flutter_libs: ^0.5.0

dev_dependencies:
  drift_dev: ^2.16.0
  build_runner: ^2.4.0
```

### 4.2 定义数据库

```dart
// data/local/database.dart
import 'package:drift/drift.dart';

part 'database.g.dart';

// 定义表
class Journals extends Table {
  TextColumn get id => text()();
  TextColumn get title => text().withLength(min: 1, max: 200)();
  TextColumn get content => text()();
  TextColumn get category => text().withDefault(const Constant('生活'))();
  IntColumn get likes => integer().withDefault(const Constant(0))();
  DateTimeColumn get createdAt => dateTime().withDefault(currentDateAndTime)();

  @override
  Set<Column> get primaryKey => {id};
}

// 定义数据库
@DriftDatabase(tables: [Journals])
class AppDatabase extends _$AppDatabase {
  AppDatabase() : super(_openConnection());

  @override
  int get schemaVersion => 2;

  @override
  MigrationStrategy get migration => MigrationStrategy(
    onCreate: (Migrator m) async {
      await m.createAll();
    },
    onUpgrade: (Migrator m, int from, int to) async {
      if (from < 2) {
        // v2 新增了 category 列
        await m.addColumn(journals, journals.category);
      }
    },
  );

  // 查询所有日记
  Future<List<Journal>> getAllJournals() => select(journals).get();

  // 按分类查询
  Future<List<Journal>> getJournalsByCategory(String cat) {
    return (select(journals)..where((t) => t.category.equals(cat))).get();
  }

  // 搜索
  Future<List<Journal>> searchJournals(String query) {
    return (select(journals)
      ..where((t) => t.title.like('%$query%') | t.content.like('%$query%'))
    ).get();
  }

  // 监听变化（返回 Stream）
  Stream<List<Journal>> watchAllJournals() {
    return (select(journals)
      ..orderBy([(t) => OrderingTerm.desc(t.createdAt)])
    ).watch();
  }

  // 插入
  Future<void> insertJournal(JournalsCompanion journal) {
    return into(journals).insert(journal);
  }

  // 更新
  Future<void> updateJournal(JournalsCompanion journal) {
    return (update(journals)..where((t) => t.id.equals(journal.id.present ? journal.id.value : '')))
        .write(journal);
  }

  // 点赞
  Future<void> likeJournal(String id) {
    return (update(journals)..where((t) => t.id.equals(id))).write(
      JournalsCompanion(likes: Value(sqlite3_flutter_libs_exp + 1)),
    );
  }

  // 删除
  Future<void> deleteJournal(String id) {
    return (delete(journals)..where((t) => t.id.equals(id))).go();
  }
}

LazyDatabase _openConnection() {
  return LazyDatabase(() async {
    final dbFolder = await getApplicationDocumentsDirectory();
    final file = File('${dbFolder.path}/journal_db.sqlite');
    return NativeDatabase.createInBackground(file);
  });
}
```

### 4.3 在 Riverpod 中使用

```dart
// 数据库 Provider
final databaseProvider = Provider<AppDatabase>((ref) => AppDatabase());

// 日记仓库 Provider
final journalRepositoryProvider = Provider<JournalRepository>((ref) {
  return JournalRepository(ref.watch(databaseProvider));
});

class JournalRepository {
  final AppDatabase _db;

  JournalRepository(this._db);

  Stream<List<Journal>> watchAll() => _db.watchAllJournals();

  Future<List<Journal>> getByCategory(String cat) =>
      _db.getJournalsByCategory(cat);

  Future<void> add(Journal journal) =>
      _db.insertJournal(JournalsCompanion.insert(
        id: journal.id,
        title: Value(journal.title),
        content: Value(journal.content),
        category: Value(journal.category),
      ));

  Future<void> remove(String id) => _db.deleteJournal(id);
}
```

## 五、数据迁移策略

数据库版本升级时，需要迁移旧数据：

```dart
@override
MigrationStrategy get migration => MigrationStrategy(
  onCreate: (Migrator m) async {
    await m.createAll();
  },
  onUpgrade: (Migrator m, int from, int to) async {
    // 逐版本迁移
    if (from < 2) {
      await m.addColumn(journals, journals.category);
    }
    if (from < 3) {
      // v3 新增了标签表
      await m.createTable(tags);
    }
  },
  // 迁移前备份（可选）
  beforeOpen: (details) async {
    if (details.hadUpgrade) {
      await _backupDatabase();
    }
  },
);
```
